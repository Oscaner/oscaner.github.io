template <typename T>
class Stack : public Vector<T> { // 由向量派生
  public: // size()、empty() 以及其他开放接口均可直接沿用
    void push(T const &e) { insert(size(), e); } // 入栈
    T pop() { return remove(size() - 1); } // 出栈
    T &top() { return (*this)[size() - 1]; } // 取顶
}; // 以向量 首/末 端为栈 底/顶
